001/**
002 * Class to model the water on the terrain
003 */
004public class Water {
005
006    /**
007     * Water 2D array to store integer depths
008     */
009    static int [][] waterDepths; // regular grid of waterValues
010    /**
011     * Constant for water depths
012     */
013    static final int waterDepthPerClick=3;
014
015
016    /**
017     * Initialise the terrain array with values of zero
018     * @param terrain the dimensions of terrain stored in object
019     */
020    static void readWaterArray(Terrain terrain){            //initialising water array
021        waterDepths= new int[terrain.dimx][terrain.dimy];
022        for (int i = 0; i < terrain.dimx ; i++) {
023            for (int j = 0; j < terrain.dimy ; j++) {
024                waterDepths[i][j]=0;
025            }
026        }
027    }
028
029
030    /**
031     * Thread safe method to increment water by a single unit
032     * @param x x coordinate of water location
033     * @param y y coordinate of water location
034     */
035    static synchronized void increaseWaterDepth(int x, int y){
036        waterDepths[x][y]=waterDepths[x][y]+1;
037    }
038
039    /**
040     * Thread safe method to decrement water by a single unit
041     * @param x x coordinate of water location
042     * @param y y coordinate of water location
043     */
044    static synchronized void decreaseWaterDepth(int x, int y){
045        waterDepths[x][y]=waterDepths[x][y]-1;
046    }
047
048    /**
049     * Method to get current surface for comparison
050     * @param x the surfaces x position
051     * @param y the surfaces y position
052     * @return current surface representing water and terrain height
053     */
054    static synchronized float getSurface(int x, int y){
055        return (float) (((waterDepths[x][y])*0.01)+Terrain.height[x][y]);
056
057    }
058
059}
